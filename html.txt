    <script>
        // 1. Функція для музики
        function toggleMusic() {
            const music = document.getElementById('bg-music');
            const btnText = document.getElementById('btn-text');
            const btnIcon = document.getElementById('btn-icon');

            if (music.paused) {
                music.play().catch(e => console.log("Музика заблокована браузером, потрібен клік"));
                btnText.innerText = "Атмосфера увімкнена";
                btnIcon.innerText = "⏸️";
            } else {
                music.pause();
                btnText.innerText = "Увімкнути атмосферу";
                btnIcon.innerText = "🎵";
            }
        }

        // 2. Дані для Вікторини
        const quizData = [
            {
                question: "Що ідеально доповнює вечір п'ятниці для Вані?",
                options: ["Просто поспати", "Юля, бокал прохолодного пива і гарний настрій", "Перегляд новин"],
                correct: 1
            },
            {
                question: "Який матеріал Ваня вважає найбільш... стильним?",
                options: ["Шовк", "Джинс", "Шкіра та латекс 🔥"],
                correct: 2
            },
            {
                question: "Найкращий напій усіх часів за версією іменинника?",
                options: ["Якісне нефільтроване", "Капучино на кокосовому", "Мінералочка"],
                correct: 0
            }
        ];

        let currentQuestion = 0;

        function loadQuiz() {
            const qBlock = document.getElementById('question-block');
            const qQuestion = document.getElementById('quiz-question');
            const qOptions = document.getElementById('quiz-options');

            if (currentQuestion < quizData.length) {
                const current = quizData[currentQuestion];
                qQuestion.innerText = current.question;
                qOptions.innerHTML = "";

                current.options.forEach((opt, index) => {
                    const btn = document.createElement('button');
                    btn.innerText = opt;
                    btn.className = "w-full text-left bg-neutral-800 hover:bg-amber-500/20 p-4 rounded-xl text-sm border border-neutral-700 transition-colors text-gray-200";
                    btn.onclick = () => checkAnswer(index);
                    qOptions.appendChild(btn);
                });
            } else {
                document.getElementById('question-block').classList.add('hidden');
                const resultBlock = document.getElementById('quiz-result');
                resultBlock.classList.remove('hidden');
                document.getElementById('quiz-feedback').innerText = "Ти знаєш Ваню на всі 100%! Тепер ти офіційно допущений до святкування!";
            }
        }

        function checkAnswer(selected) {
            if (selected === quizData[currentQuestion].correct) {
                alert("Правильно! 🎉");
                currentQuestion++;
                loadQuiz();
            } else {
                alert("Хмм, спробуй ще раз! Ваня б такого не вибрав 😜");
            }
        }

        // Ініціалізація квізу при завантаженні
        loadQuiz();

        // 3. Функція для відкриття секретного побажання
        function openSecret() {
            const content = document.getElementById('secret-content');
            const trigger = document.getElementById('beer-trigger');
            const label = document.getElementById('trigger-label');

            // Зміна іконки на відкрите пиво
            trigger.innerText = "🍻";
            trigger.classList.remove('animate-pulse');
            label.innerText = "Відкорковано!";

            // Поява блоку з текстом
            content.classList.remove('hidden');
            setTimeout(() => {
                content.classList.remove('opacity-0', 'translate-y-4');
                content.classList.add('opacity-100', 'translate-y-0');
            }, 50);
        }
    </script>
